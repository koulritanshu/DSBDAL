var data = sc.textFile("textFile");
data.collect;
var splitdata = data.flatMap(line => line.split(" "));
splitdata.collect;
var mapdata = splitdata.map(word => (word,1));
mapdata.collect;
var reducedata = mapdata.reduceByKey(_+_);
reducedata.collect;


to save as a file then do
reducedata.saveAsTextFile("output.txt")
reducedata.foreach(println)
